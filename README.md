<a name="section-docs-readme-02-usage-md"></a>
Usage
----

```javascript
var furFonts = require('fur-fonts'),
    fs = require('fs');


var fontA = furFonts.a(),
    fontB = furFonts.b();

console.log(fontA);
console.log(fontB);
```
